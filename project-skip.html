def rk4_integrator(x_0, t_0, t_n, dt, dx, u):
    x = x_0
    t = t_0
    
    yield (x, t)
    
    while t <= t_n:
        # eval x_k+1 using x_k and t_k
        k1 = dx(x, t, u)
        k2 = dx(x + dt * k1 / 2, t + dt / 2, u)
        k3 = dx(x + dt * k2 / 2, t + dt / 2, u)
        k4 = dx(x + dt * k3, t + dt, u)
        
        x = x + dt * (k1 + 2*k2 + 2*k3 + k4) * (1/6)
        
        # t increament to t_k+1
        t = t + dt
        
        yield (x, t)
